00057 LM_FORCE_INLINE int TriAccel::Load(const Math::Vec3& A, const Math::Vec3& B, const Math::Vec3& C)
00058 {
00059     static const int waldModulo[4] = { 1, 2, 0, 1 };
00060 
00061     Math::Vec3 b = C - A;
00062     Math::Vec3 c = B - A;
00063     Math::Vec3 N = Math::Cross(c, b);
00064 
00065     k = 0;
00066     // Determine the largest projection axis
00067     for (int j = 0; j < 3; j++)
00068     {
00069         if (Math::Abs(N[j]) > Math::Abs(N[k]))
00070         {
00071             k = j;
00072         }
00073     }
00074 
00075     uint32_t u = waldModulo[k];
00076     uint32_t v = waldModulo[k+1];
00077     const Math::Float n_k = N[k];
00078     const Math::Float denom = b[u]*c[v] - b[v]*c[u];
00079 
00080     if (denom == 0)
00081     {
00082         k = 3;
00083         return 1;
00084     }
00085 
00086     // Pre-compute intersection calculation constants
00087     n_u   =  N[u] / n_k;
00088     n_v   =  N[v] / n_k;
00089     n_d   =  Math::Dot(Math::Vec3(A), N) / n_k;
00090     b_nu  =  b[u] / denom;
00091     b_nv  = -b[v] / denom;
00092     a_u   =  A[u];
00093     a_v   =  A[v];
00094     c_nu  =  c[v] / denom;
00095     c_nv  = -c[u] / denom;
00096 
00097     return 0;
00098 }

00100 LM_FORCE_INLINE bool TriAccel::Intersect(const Ray& ray, Math::Float mint, Math::Float maxt, Math::Float& u, Math::Float& v, Math::Float& t) const
00101 {
00102     Math::Float o_u, o_v, o_k, d_u, d_v, d_k;
00103     switch (k)
00104     {
00105     case 0:
00106         o_u = ray.o[1];
00107         o_v = ray.o[2];
00108         o_k = ray.o[0];
00109         d_u = ray.d[1];
00110         d_v = ray.d[2];
00111         d_k = ray.d[0];
00112         break;
00113 
00114     case 1:
00115         o_u = ray.o[2];
00116         o_v = ray.o[0];
00117         o_k = ray.o[1];
00118         d_u = ray.d[2];
00119         d_v = ray.d[0];
00120         d_k = ray.d[1];
00121         break;
00122 
00123     case 2:
00124         o_u = ray.o[0];
00125         o_v = ray.o[1];
00126         o_k = ray.o[2];
00127         d_u = ray.d[0];
00128         d_v = ray.d[1];
00129         d_k = ray.d[2];
00130         break;
00131 
00132     default:
00133         return false;
00134     }
00135 
00136 
00137 #if LM_DEBUG_MODE
00138     if (d_u * n_u + d_v * n_v + d_k == 0)
00139     {
00140         return false;
00141     }
00142 #endif
00143 
00144     // Calculate the plane intersection (Typo in the thesis?)
00145     t = (n_d - o_u*n_u - o_v*n_v - o_k) / (d_u * n_u + d_v * n_v + d_k);
00146     if (t < mint || t > maxt)
00147     {
00148         return false;
00149     }
00150 
00151     // Calculate the projected plane intersection point
00152     const Math::Float hu = o_u + t * d_u - a_u;
00153     const Math::Float hv = o_v + t * d_v - a_v;
00154 
00155     // In barycentric coordinates
00156     u = hv * b_nu + hu * b_nv;
00157     v = hu * c_nu + hv * c_nv;
00158 
00159     return u >= 0 && v >= 0 && u+v <= 1;
00160 }
